# Test cases for the say_my_name function

def say_my_name(first_name, last_name=""):
    """
    function that prints My name is <first name> <last name>
    first_name and last_name must be strings otherwise, a TypeError exception is raised
    """
    if not isinstance(first_name, str):
        raise TypeError("first_name must be a string")
    if not isinstance(last_name, str):
        raise TypeError("last_name must be a string")
    print (f"My name is {first_name} {last_name}")

# Test case 1: first_name must be a string (should raise TypeError)
>>> say_my_name("John")

	...

# Test case 2: last_name must be a string (should raise TypeError)
>>> say_my_name("Smith")
Traceback (most recent call last):
        ...
TypeError: last_name must be a string

# Test case 3: both first_name and last_name are strings (should print correctly)
say_my_name("Walter", "White")

# Test case 4: empty string as last_name (should print correctly)
say_my_name("Bob", "")

# Test case 5: both first_name and last_name are None (should print correctly)
say_my_name(None, None)
